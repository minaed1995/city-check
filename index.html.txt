  <script>
    // لیست ثابت شهرها (می‌توانید این لیست را بروزرسانی کنید)
    const cities = [
      "تهران", "مشهد", "اصفهان", "تبریز", "شیراز", "اهواز", "کرج", "قم", "کرمانشاه", "ارومیه",
      "زنجان", "رشت", "کرمان", "یزد", "ساری", "بندرعباس", "همدان", "بوشهر", "خرم‌آباد", "سنندج"
    ];

    const resultsEl = document.getElementById('results');
    const qInput = document.getElementById('q');

    function normalize(s){ return (s||'').trim().toLocaleLowerCase(); }

    function search(){
      const query = normalize(qInput.value);
      resultsEl.innerHTML = '';
      if(!query){ return; }
      const found = cities.filter(c => normalize(c) === query);
      if(found.length){
        resultsEl.innerHTML = `<li class="city found">${found[0]} — ارسال انجام می‌شود</li>`;
      } else {
        resultsEl.innerHTML = `<li class="city notfound">متأسفانه ارسال نداریم</li>`;
      }
    }

    document.getElementById('btnSearch').addEventListener('click', search);
    qInput.addEventListener('keydown', e => { if(e.key==='Enter'){ search(); } });
  </script>
